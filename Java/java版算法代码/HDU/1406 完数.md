import java.util.Scanner;
public class Main {
    public static void main(String[] args){
        Scanner input = new Scanner(System.in);
        int T = input.nextInt();
        int num1,num2,cnt,sum;
        while(T!=0){
            T--;
            num1 = input.nextInt();
            num2 = input.nextInt();
            if(num1>num2){
                int temp = num1;
                num1 = num2;
                num2 = temp;
            }
            cnt = 0;
            for(int i=num1;i<=num2;i++){
                sum = 1;
                for(int j=2;j<=Math.sqrt(i);j++){
                    if(i%j==0){
                        sum += (j+i/j);
                    }
                }
                if(sum==i){
                    cnt++;
                }
            }
            System.out.println(cnt);
        }
    }
}
