import java.util.Scanner;
public class Main {
    public static void main(String[] args){
        int m, n, flag, g, s, b;
        Scanner input = new Scanner(System.in);
        while(input.hasNextInt()){
            m = input.nextInt();
            n = input.nextInt();
            flag = 0;
            for(int i=m;i<=n;i++) {
                b = i / 100;
                s = i / 10 % 10;
                g = i % 10;
                if(i == b*b*b+s*s*s+g*g*g){
                    if(flag == 0){
                        System.out.print(i);
                        flag++;
                    }
                    else{
                        System.out.print(" "+i);
                    }
                }
            }
            if(flag==0){
                System.out.print("no");
            }
            System.out.println();
        }
    }
}
